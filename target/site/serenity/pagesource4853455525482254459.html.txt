            navigationLog = [];
        }
        var currentTime = (new Date()).toISOString().slice(0, 19).replace('T', ' ');
        var latestLog = sessionStorage.getItem("navigation_log_latest");
        if (!latestLog) {
            latestLog = localStorage.getItem("navigation_log_latest");//NOSONAR
        }
        latestLog = JSON.parse(latestLog);
        navigationLog.push({from: latestLog.to, to: '/auth/login', time: currentTime});

        var date = new Date();
